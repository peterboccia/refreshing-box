The refreshing-box demo

Simple use:


	$('.refreshing-box').refreshingBox({
	  description: 'Aggiorna automaticamente',
	  table: true,
	  checkBoxId: '',
	  interval: 1000,
	  //checked: true,
	  onRefresh: function() {
		$('#console').append(new Date() + '<br/>');
	  },
	  onStart: function() {
		$('#console').append('started' + '<br/>');
	  },
	  onStop: function() {
		$('#console').append('stopped' + '<br/>');
	  }
	});
